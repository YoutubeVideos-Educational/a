EXPERIMENT 6 — BISON + FLEX
Step 1
nano calc.y
Step 2 — code
%{
#include <stdio.h>
#include <stdlib.h>
%}

%token NUMBER

%%

expr: expr '+' expr { printf("%d\n",$1+$3); }
    | expr '-' expr { printf("%d\n",$1-$3); }
    | expr '*' expr { printf("%d\n",$1*$3); }
    | expr '/' expr { printf("%d\n",$1/$3); }
    | NUMBER { $$=$1; }
    ;

%%

int main()
{
    yyparse();
    return 0;
}

int yyerror()
{
    return 0;
}

Save.

Step 3
nano calc.l
Step 4 — code
%{
#include "y.tab.h"
%}

%%

[0-9]+ { yylval=atoi(yytext); return NUMBER; }

[+\-*/] { return yytext[0]; }

[ \t] ;

\n return 0;

%%

int yywrap()
{
    return 1;
}

Save.

Step 5 — run
bison -d calc.y
flex calc.l
gcc y.tab.c lex.yy.c -lfl -o calc
./calc
