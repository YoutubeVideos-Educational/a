Step 3 — paste code
#include<stdio.h>
#include<ctype.h>
#include<string.h>

char prod[10][10];
char first[10], follow[10];
int n;

void findFirst(char c,int k)
{
    int i;

    if(!isupper(c))
    {
        first[k++] = c;
        return;
    }

    for(i=0;i<n;i++)
    {
        if(prod[i][0]==c)
        {
            if(!isupper(prod[i][2]))
                first[k++] = prod[i][2];
            else
                findFirst(prod[i][2],k);
        }
    }
}

void findFollow(char c)
{
    int i,j;

    if(prod[0][0]==c)
        follow[0]='$';

    for(i=0;i<n;i++)
    {
        for(j=2;j<strlen(prod[i]);j++)
        {
            if(prod[i][j]==c)
            {
                if(prod[i][j+1]!='\0')
                    follow[0]=prod[i][j+1];
            }
        }
    }
}

int main()
{
    int i;
    char c;

    printf("Enter number of productions: ");
    scanf("%d",&n);

    printf("Enter productions (E=TR):\n");

    for(i=0;i<n;i++)
        scanf("%s",prod[i]);

    printf("Find FIRST of: ");
    scanf(" %c",&c);

    findFirst(c,0);

    printf("FIRST = ");
    for(i=0; first[i]!='\0'; i++)
        printf("%c ",first[i]);

    printf("\n");

    printf("Find FOLLOW of: ");
    scanf(" %c",&c);

    findFollow(c);

    printf("FOLLOW = ");
    for(i=0; follow[i]!='\0'; i++)
        printf("%c ",follow[i]);

    return 0;
}
